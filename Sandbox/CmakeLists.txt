cmake_minimum_required(VERSION 3.10)

project(Sandbox)

set (CMAKE_CXX_STANDARD 23)
set (CMAKE_CXX_STANDARD_REQUIRED ON)

add_executable(Sandbox $<$<CONFIG:Dist>:WIN32>
src/Sandbox.cpp
)

target_link_libraries(Sandbox PRIVATE Lemon)

if(${WIN32})
add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
  COMMAND ${CMAKE_COMMAND} -E copy -t $<TARGET_FILE_DIR:${PROJECT_NAME}> $<TARGET_RUNTIME_DLLS:${PROJECT_NAME}>
  COMMAND_EXPAND_LISTS
)
endif()

# Collect all assets
file(GLOB_RECURSE ASSET_FILES CONFIGURE_DEPENDS "assets/*")

foreach(ASSET ${ASSET_FILES})
    file(RELATIVE_PATH REL_PATH "${CMAKE_SOURCE_DIR}/assets" "${ASSET}")
    set(OUTPUT "${CMAKE_BINARY_DIR}/assets/${REL_PATH}")

    add_custom_command(
        OUTPUT  "${OUTPUT}"
        COMMAND ${CMAKE_COMMAND} -E copy_if_different "${ASSET}" "${OUTPUT}"
        DEPENDS "${ASSET}"
        COMMENT "Copying ${REL_PATH}"
    )
    list(APPEND COPIED_ASSETS "${OUTPUT}")
endforeach()

add_custom_target(CopyAssets ALL DEPENDS ${COPIED_ASSETS})
add_dependencies(Sandbox CopyAssets)
